-- Row Number by Sales
SELECT
    "Product Name",
    Sales,
    ROW_NUMBER() OVER
    (
        ORDER BY Sales DESC
    ) AS Row_Num
FROM superstore
LIMIT 20;

-- Row Number by Customer Sales
SELECT
    "Customer Name",
    Total_Sales,
    ROW_NUMBER() OVER (
        ORDER BY Total_Sales DESC
    ) AS Row_Num
FROM
(
    SELECT
        "Customer Name",
        ROUND(SUM(Sales),2) AS Total_Sales
    FROM superstore
    GROUP BY "Customer Name"
) AS CustomerSales;

-- Rank Customers by Total Sales
SELECT
    "Customer Name",
    Total_Sales,
    RANK() OVER (
        ORDER BY Total_Sales DESC
    ) AS Sales_Rank
FROM
(
    SELECT
        "Customer Name",
        ROUND(SUM(Sales),2) AS Total_Sales
    FROM superstore
    GROUP BY "Customer Name"
) AS CustomerSales;

-- Dense Rank Customers by Total Sales

SELECT
    "Customer Name",
    Total_Sales,
    DENSE_RANK() OVER (
        ORDER BY Total_Sales DESC
    ) AS Sales_Rank
FROM
(
    SELECT
        "Customer Name",
        ROUND(SUM(Sales),2) AS Total_Sales
    FROM superstore
    GROUP BY "Customer Name"
) AS CustomerSales;
